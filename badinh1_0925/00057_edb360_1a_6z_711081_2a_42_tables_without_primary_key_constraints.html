select  OWNER, TABLE_NAME
from    dba_tables dt
where   not exists (
        select  'TRUE'
        from    dba_constraints dc
        where   dc.TABLE_NAME = dt.TABLE_NAME
        and dc.owner = dt.owner
        and     dc.CONSTRAINT_TYPE='P')

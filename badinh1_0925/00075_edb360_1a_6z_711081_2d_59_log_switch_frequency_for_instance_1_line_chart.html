WITH
pivot1 AS (
select
s.INSTANCE_NUMBER,
s.snap_id,
       TO_CHAR(MIN(s.begin_interval_time), 'YYYY-MM-DD HH24:MI:SS') begin_time,
       TO_CHAR(MIN(s.end_interval_time), 'YYYY-MM-DD HH24:MI:SS') end_time
from dba_hist_snapshot s where s.INSTANCE_NUMBER=1
and snap_id BETWEEN 14504+1 AND 14709
AND (CAST(s.end_interval_time AS DATE) - CAST(s.begin_interval_time AS DATE)) * 24 * 60 > 1 -- ignore snaps closer than 1m appart
group by s.snap_id, s.INSTANCE_NUMBER
),
pivot2 AS (
select e.snap_id,e.instance_number, 'log switches  (derived)',
(e.sequence#  - b.sequence#) log_switches
from   dba_hist_thread e, dba_hist_thread b
where e.snap_id BETWEEN 14504+1 AND 14709
     and e.snap_id  = b.snap_id+1
     and b.dbid  =  2417711081
     and e.dbid  =  2417711081
and b.instance_number  = 1
and e.instance_number  = 1
and b.thread#  = e.thread#
and b.thread_instance_number  = e.thread_instance_number and e.thread_instance_number  = 1
)
select
p.snap_id,
       p.begin_time,
       p.end_time,
       q.log_switches,
       0 dummy_02,
       0 dummy_03,
       0 dummy_04,
       0 dummy_05,
       0 dummy_06,
       0 dummy_07,
       0 dummy_08,
       0 dummy_09,
       0 dummy_10,
       0 dummy_11,
       0 dummy_12,
       0 dummy_13,
       0 dummy_14,
       0 dummy_15
from pivot1 p left outer join pivot2 q on (p.snap_id=q.snap_id and p.instance_number=q.instance_number)
order by p.snap_id;

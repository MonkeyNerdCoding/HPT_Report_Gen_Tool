WITH
totals AS (
SELECT /*+  MATERIALIZE NO_MERGE   DYNAMIC_SAMPLING(4)  */
       /* 3h.77 */
       con_id,
       sql_id,
       SUM(DIRECT_WRITES_DELTA) direct_writes
  FROM dba_hist_sqlstat
 WHERE snap_id BETWEEN 14504 AND 14709
   AND dbid = 2417711081
 GROUP BY
       con_id,
       sql_id
 ORDER BY
       SUM(DIRECT_WRITES_DELTA) DESC NULLS LAST
),
filter AS(
select * from totals where ROWNUM < 16
)
SELECT /*+  NO_MERGE  */ /* 3h.77 */
       s.sql_id,
       s.direct_writes,
       DBMS_LOB.SUBSTR(t.sql_text, 1000) sql_text
  FROM filter s, dba_hist_sqltext t
 WHERE t.sql_id(+) = s.sql_id
   AND t.dbid(+) = 2417711081
   AND t.con_id(+) = s.con_id
 ORDER BY s.direct_writes DESC NULLS LAST;

WITH
dba_hist_tbspc_space_usage_sqf AS (
SELECT /*+
       MATERIALIZE
       NO_MERGE
       FULL(h.INT$DBA_HIST_TBSPC_SPACE_USAGE.sn)
       FULL(h.INT$DBA_HIST_TBSPC_SPACE_USAGE.tb)
       USE_HASH(h.INT$DBA_HIST_TBSPC_SPACE_USAGE.sn h.INT$DBA_HIST_TBSPC_SPACE_USAGE.tb)
       FULL(h.sn)
       FULL(h.tb)
       USE_HASH(h.sn h.tb)
       */
       h.snap_id,
       h.tablespace_id,
       h.tablespace_size
  FROM dba_hist_tbspc_space_usage h
 WHERE h.snap_id >= 14504 and h.snap_id <= 14709
   AND h.dbid = 2417711081
),
dba_hist_snapshot_sqf AS (
SELECT /*+
       MATERIALIZE
       NO_MERGE
       FULL(s.INT$DBA_HIST_SNAPSHOT.WRM$_SNAPSHOT)
       FULL(s.WRM$_SNAPSHOT)
       */
       s.snap_id,
       s.end_interval_time
  FROM dba_hist_snapshot s
 WHERE s.snap_id >= 14504
   AND s.dbid = 2417711081
   AND s.instance_number = 1
)
SELECT /*+ USE_HASH(h s) */
       t.tablespace_name,
       TO_CHAR(s.end_interval_time, 'YYYY-MM-DD') end_day,
       ROUND(h.tablespace_size * t.block_size / POWER(10,6), 3) value_mb
  FROM dba_hist_tbspc_space_usage_sqf h,
       dba_hist_snapshot_sqf          s,
       v$tablespace                   v,
       dba_tablespaces                t
 WHERE s.snap_id         = h.snap_id
   AND v.ts#             = h.tablespace_id
   AND t.tablespace_name = v.name
   AND s.end_interval_time IN (
SELECT /*+ USE_HASH(h s) */
       MAX(s.end_interval_time) end_time
  FROM dba_hist_tbspc_space_usage_sqf h,
       dba_hist_snapshot_sqf          s,
       v$tablespace                   v,
       dba_tablespaces                t
 WHERE s.snap_id         = h.snap_id
   AND v.ts#             = h.tablespace_id
   AND t.tablespace_name = v.name
GROUP BY
       h.tablespace_id,
       t.tablespace_name,
       TO_CHAR(s.end_interval_time, 'YYYY-MM-DD')
)
ORDER BY t.tablespace_name, TO_CHAR(s.end_interval_time, 'YYYY-MM-DD');
